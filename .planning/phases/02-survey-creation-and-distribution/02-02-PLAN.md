    Create `src/lib/services/token.service.ts`:
    ```typescript
    import { randomBytes } from 'crypto';
    import { readRows, appendRow } from './csv.service';
    import type { Token } from '@/lib/types';

    function tokenFile(surveyId: string) {
      return `tokens-${surveyId}.csv`;
    }

    export async function generateToken(surveyId: string, email: string): Promise<string> {
      const filename = tokenFile(surveyId);
      const existing = await readRows<Record<string, string>>(filename);
      const found = existing.find(r => r.email === email && r.surveyId === surveyId);
      if (found) return found.token;

      const token = randomBytes(32).toString('hex');
      await appendRow(filename, {
        token,
        email,
        surveyId,
        status: 'pending',
        createdAt: new Date().toISOString(),
        submittedAt: '',
      });
      return token;
    }

    export async function validateToken(token: string, surveyId: string): Promise<Token | null> {
      const filename = tokenFile(surveyId);
      const rows = await readRows<Record<string, string>>(filename);
      const row = rows.find(r => r.token === token && r.surveyId === surveyId);
      if (!row || row.status !== 'pending') return null;
      return {
        token: row.token,
        surveyId: row.surveyId,
        email: row.email,
        used: row.status === 'submitted',
        sentAt: row.createdAt,
      };
    }
    ```

    import nodemailer from 'nodemailer';

    export function createTransporter(settings: SmtpSettings) {
      return nodemailer.createTransport({
        host: settings.host,
        port: Number(settings.port),
        secure: Number(settings.port) === 465,
        auth: { user: settings.username, pass: settings.password },
        connectionTimeout: 15_000,
        socketTimeout: 15_000,
      });
    }

    export async function testSmtpConnection(settings: SmtpSettings): Promise<void> {
      const transporter = createTransporter(settings);
      await transporter.verify();
    }

    **Create `src/app/api/surveys/[id]/invite/route.ts`:**
    POST endpoint that:
    1. Checks auth (iron-session)
    2. Parses body: `{ emails: string[], locale?: string }` — locale defaults to 'en'
    3. Reads SMTP settings; returns 400 `{ error: 'SMTP not configured' }` if null
    4. Validates each email against basic regex `/^[^\s@]+@[^\s@]+\.[^\s@]+$/`; collects invalid list
    5. Deduplicates valid emails
    6. Gets survey via `getSurvey(id)` — returns 404 if not found
    7. Creates nodemailer transporter from SMTP settings
    8. For each email: calls `generateToken(surveyId, email)`, then `sendInvitation({...})`
    9. Tracks results: `{ email, status: 'sent' | 'failed' | 'already-invited', error?: string }`
    10. Already-invited detection: if generateToken returns a token that already existed (check tokens CSV before and after) — mark as 'already-invited' in results; still send email (re-sending invitation is valid behavior per requirements — "re-inviting same email reuses existing token")
    11. Returns 200 `{ results: SendResult[], successCount: number, failCount: number }`
    12. Send is sequential (not parallel) — `for...of` loop, NOT `Promise.all`
